<script>
    function toggleWindow() {
        const win = document.getElementById('wallWindow');
        win.style.display = (win.style.display === 'block') ? 'none' : 'block';
    }

    function calculateWall() {
        // მონაცემების წამოღება
        const length = parseFloat(document.getElementById('length').value);
        const height = parseFloat(document.getElementById('height').value);
        const materialSize = parseFloat(document.getElementById('material').value);
        const unitPrice = parseFloat(document.getElementById('unitPrice').value) || 0;

        if (!length || !height) {
            alert("გთხოვთ შეავსოთ ზომები");
            return;
        }

        // 1. ფართობი
        const area = length * height;
        
        // 2. მასალის რაოდენობა (დაახლოებით 5% მარაგით)
        const count = Math.ceil(area / materialSize * 1.05);
        const materialTotalPrice = count * unitPrice;

        // 3. ხსნარის გათვლა (საშუალო მაჩვენებელი 1მ2-ზე)
        // ბლოკისთვის საჭიროა დაახლ. 0.02 კუბი ხსნარი 1მ2-ზე
        const sandVolume = (area * 0.025).toFixed(2); 
        const cementBags = Math.ceil(area * 0.25); // დაახლ. 1 მეშოკი 4 კვადრატზე

        const sandPrice = sandVolume * 55;
        const cementPrice = cementBags * 17;
        const totalMortarPrice = sandPrice + cementPrice;

        const totalSum = materialTotalPrice + totalMortarPrice;

        // შედეგების გამოტანა
        document.getElementById('resArea').innerText = area.toFixed(2);
        document.getElementById('resCount').innerText = count;
        document.getElementById('resMaterialPrice').innerText = materialTotalPrice.toFixed(2);
        document.getElementById('resCement').innerText = cementBags;
        document.getElementById('resSand').innerText = sandVolume;
        document.getElementById('resMortarPrice').innerText = totalMortarPrice.toFixed(2);
        document.getElementById('resTotal').innerText = totalSum.toFixed(2);

        document.getElementById('results').style.display = 'block';
    }
</script>
