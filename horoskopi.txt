<script>
    function autoDiam() {
        const circ = document.getElementById('circ').value;
        const diam = circ / Math.PI;
        document.getElementById('q_mid').value = diam.toFixed(1);
    }

    function calcQvevri() {
        const d = document.getElementById('q_top').value / 2;
        const D = document.getElementById('q_mid').value / 2;
        const h = document.getElementById('q_height').value;
        if(!D || !h) return alert("შეავსეთ მონაცემები!");

        const v = ((Math.PI * h) / 6) * (3 * Math.pow(D, 2) + 3 * Math.pow(d, 2) + Math.pow(h, 2) / 4); // მიახლოებითი ფორმულა
        const total = Math.round(v / 1000 * 0.85); // ქვევრის ფორმის კორექცია
        
        document.getElementById('q_res_box').style.display = "block";
        document.getElementById('q_res').innerText = total + " ლიტრი";
        document.getElementById('grapes_needed').innerText = "საჭიროა ~ " + Math.round(total * 1.4) + " კგ ყურძენი";
        document.getElementById('chacha_res').innerText = "ნარჩენი ჭაჭა: ~ " + Math.round(total * 0.4) + " კგ";
    }

    function calcAlcohol() {
        const sugar = parseFloat(document.getElementById('sugar').value);
        const plus = parseFloat(document.getElementById('target_plus').value) || 0;
        if(!sugar) return alert("შეიყვანეთ შაქრიანობა!");

        const alc = (sugar * 0.59).toFixed(1);
        document.getElementById('alc_res').style.display = "block";
        document.getElementById('alc_val').innerText = alc + "% ალკოჰოლი";
        
        if(plus > 0) {
            document.getElementById('sugar_add_info').innerText = "მატებისთვის საჭიროა " + (plus * 17) + "გ შაქარი/ლიტრზე.";
        } else {
            document.getElementById('sugar_add_info').innerText = "";
        }
    }
</script>
